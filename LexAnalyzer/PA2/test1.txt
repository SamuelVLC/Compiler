-- Teste 1: CLASS Token
-- Input:
class Main {
    feature: method(): Object {
        let x: Int <- 10 + 5;
        if true then
            "True"
        else
            "False"
    }
}

-- Resultado Esperado:
-- A token CLASS deve ser reconhecida corretamente

-- Teste 2: Identificadores Tipo e Objeto
-- Input:
int I
Interger i

-- Resultado Esperado:
-- Deve ser reconhecido TYPEID (int) seguido por OBJECTID (I)
-- Deve ser reconhecido TYPEID (Interger) seguido por OBJECTID (i)

-- Teste 3: Tokens IF e THEN
-- Input:
IFIF THEN
IF THEN
THEN

-- Resultado Esperado:
-- Deve ser reconhecido IF seguido por IF
-- Deve ser reconhecido IF seguido por THEN
-- Deve ser reconhecido THEN

-- Teste 4: Tokens NEW e ELSE
-- Input:
NEW
new
ELSE

-- Resultado Esperado:
-- Deve ser reconhecido NEW
-- Deve ser reconhecido OBJECTID (new)
-- Deve ser reconhecido ELSE

-- Teste 5: Tokens CONST e IF
-- Input:
CONST IF

-- Resultado Esperado:
-- Deve ser reconhecido TYPEID (CONST) seguido por IF

-- Teste 6: Ignorar Token LET_STMT
-- Input:
LET_STMT

-- Resultado Esperado:
-- A token LET_STMT deve ser ignorada e não retornar nenhum token correspondente



-- Teste 7: Ignorar Token error
